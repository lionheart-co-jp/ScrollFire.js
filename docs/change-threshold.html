JavaScript

var targets = document.querySelectorAll('.block');

var fire = new ScrollFire();
fire.addTrigger(targets, function(el) {
    el.classList.add('block-view');
}, function(el) {
    el.classList.remove('block-view');
}, { ratio: 15 });
fire.start();
